#include <TinyGPS++.h>
#include <SoftwareSerial.h>
TinyGPSPlus gps;
SoftwareSerial ss(4, 3); 
void setup() {
  Serial.begin(9600);
  ss.begin(9600);
}
void loop() {
  while (ss.available() > 0)
    gps.encode(ss.read());
  if (gps.location.isUpdated()) {
    Serial.print("Lat: "); Serial.println(gps.location.lat(), 6);
    Serial.print("Lng: "); Serial.println(gps.location.lng(), 6);
  }
}
